\section*{프로그램 C의 정적분석 결과}
Project1과 대부분 동일하다. 다만 Parity 도메인 대신 `1867로 나눈 나머지'를 나타내는 Reminder 도메인을 사용한다.

\subsection*{1. Definition of Domains}
\begin{align*}
  \hat{n} \in \hat{\mathbb{Z}}            = & \quad (\hat{Reminder} \times \hat{Interval}) \\
  \hat{r} \in \hat{Reminder}    = & \quad \{ \bot, \top \} \cup \{ 0, ... , 1866 \}
\end{align*}


\subsection*{2. 갈로아 연결}
\begin{align*}
  2^{\mathbb{Z}} \stackrel{{\alpha}_{\mathbb{Z}}}{\longrightarrow} & \quad \hat{\mathbb{Z}} \\
  \alpha_{\mathbb{Z}} X = & \quad (\alpha_r X, \alpha_i X)
\end{align*}

\begin{align*}
  \alpha_{r} X =  & \quad X\textrm{의 모든 원소를 1867로 나눈 나머지가 } n(\in [0, 1866]) \textrm{이면 } n, \\
  & \quad \textrm{그 외의 경우 } \top
\end{align*}


\subsection*{3. Definition of $\hat{+}, \hat{-}, \hat{\times}$ }
\subsubsection*{Definition of $\hat{+}$}
\begin{align*}
  % \hat{+_r}
  \bot \,\, \hat{+}_r \,\, \hat{r} = & \quad \bot \\
  \hat{r} \,\, \hat{+}_r \,\, \bot = & \quad \bot \\
  \top \,\, \hat{+}_r \,\, \hat{r} = & \quad \top \\
  \hat{r} \,\, \hat{+}_r \,\, \top = & \quad \top \\
  \hat{r_1} \,\, \hat{+}_r \,\, \hat{r_2} = & \quad (\hat{r_1} + \hat{r_2})\textrm{의 나머지} \\
\end{align*}

\subsubsection*{Definition of $\hat{-}$}
\begin{align*}
  \hat{-}_r \,\, \bot = & \bot \\
  \hat{-}_r \,\, \top = & \top \\
  \hat{-}_r \,\, \hat{r} = & (1867 - \hat{r})
\end{align*}

\subsubsection*{Definition of $\hat{\times}$}
\begin{align*}
  % \hat{+_r}
  \bot \,\, \hat{\times}_r \,\, \hat{r} = & \quad \bot \\
  \hat{r} \,\, \hat{\times}_r \,\, \bot = & \quad \bot \\
  \top \,\, \hat{\times}_r \,\, \hat{r} = & \quad \top \\
  \hat{r} \,\, \hat{\times}_r \,\, \top = & \quad \top \\
  \hat{r_1} \,\, \hat{\times}_r \,\, \hat{r_2} = & \quad (\hat{r_1} \times \hat{r_2})\textrm{의 나머지} \\
\end{align*}
